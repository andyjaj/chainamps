195  inline TranslationBlock<T> contract_conditional(const MPX_matrix& A, bool conjA,const MPX_matrix& B,bool conjB,const std::vector<MPXPair>& contractidxs,const std::pair<const std::vector<MPXInt>,const std::vector<MPXInt> >& condition){
196  //first check spectra match
197  if (A.m_SpectrumPtr!=B.m_SpectrumPtr){std::cout <<"Physical Indices don't match!" << std::endl;exit(1);}
198  //next check contraction indices match sizes match!
199  for (std::vector<MPXPair >::const_iterator cit=contractidxs.begin();cit!=contractidxs.end();++cit){
200  if (!match(A.m_Indices.at(cit->first),conjA,B.m_Indices.at(cit->second),conjB)){exit(1);}
201  }
202  //make list of old index order (0 to nA-1 and 0 to nB-1)
203  std::vector<Sparseint> neworderA(A.m_Indices.size());
204  std::iota(neworderA.begin(),neworderA.end(),0);
205  std::vector<Sparseint> neworderB(B.m_Indices.size());
206  std::iota(neworderB.begin(),neworderB.end(),0);
207  //shift the condition indices to the right
208  if (condition.first.begin()==condition.first.end() || condition.second.begin()==condition.second.end()){
209  std::cout << "Not enough condition indices?" << std::endl; exit(1);
210  }
211  std::vector<MPXInt> Adims=A.dimsvector();
212  std::vector<MPXInt> Bdims=B.dimsvector();
213  if (contractidxs.begin()==contractidxs.end()){
214  std::cout << "Not enough contraction indices?" << std::endl; exit(1);
215  }
216  Sparseint AConditionColumns=1;
217  for (std::vector<MPXInt>::const_iterator cit=condition.first.begin();cit!=condition.first.end();++cit){
218  *(std::remove(neworderA.begin(),neworderA.end(),abs(*cit)))=abs(*cit);
219  AConditionColumns*=Adims.at(*cit);
220  }
221  Sparseint BConditionColumns=1;
222  for (std::vector<MPXInt>::const_iterator cit=condition.second.begin();cit!=condition.second.end();++cit){
223  *(std::remove(neworderB.begin(),neworderB.end(),abs(*cit)))=abs(*cit);
224  BConditionColumns*=Bdims.at(*cit);
225  }
226  //shift contraction indices to the right
227  Sparseint ContractColumns=1; //there's got to be at least one?
228  for (std::vector<MPXPair>::const_iterator cit=contractidxs.begin();cit!=contractidxs.end();++cit){
229  *(std::remove(neworderA.begin(),neworderA.end(),cit->first))=cit->first;
230  *(std::remove(neworderB.begin(),neworderB.end(),cit->second))=cit->second;
231  ContractColumns*=Adims.at(cit->first);
232  //container size doesn't change!
233  }
234  //reshape sparsematrices
235  Sparseint Anewrowidxnum=A.m_Indices.size()-contractidxs.size()-condition.first.size();
236  Sparseint Bnewrowidxnum=B.m_Indices.size()-contractidxs.size()-condition.second.size();
237  SparseMatrix Ar(reshape(A.m_Matrix,A.m_NumRowIndices,Anewrowidxnum,Adims,neworderA,conjA));
238  SparseMatrix Br(reshape(B.m_Matrix,B.m_NumRowIndices,Bnewrowidxnum,Bdims,neworderB,conjB));
239  SparseMatrix fullsparse(Ar.rows()*Br.rows(),AConditionColumns*BConditionColumns);
240  std::vector<MPXInt> extractioncolumns;
241  //loop over sub columns of B, decompose into condition indices and remaining contraction indices.
242  for (Sparseint Bcdcol=0;Bcdcol<BConditionColumns;++Bcdcol){
243  //find charges corresponding to Bcdcol
244  State BCondState(B.CombinationState(condition.second,Bcdcol));
245  for (Sparseint Acdcol=0;Acdcol<AConditionColumns;++Acdcol){
246  //complete the condition check, and sum over contraction indices if check is true
247  if (BCondState==-A.CombinationState(condition.first,Acdcol)){
248  extractioncolumns.push_back(Acdcol+AConditionColumns*Bcdcol);
249  for (Sparseint contractcol=0; contractcol<ContractColumns;++contractcol){
250  //form the actual column of Br
251  Sparseint Bcol=Bcdcol+BConditionColumns*contractcol;
252  if (Br.get_p(Bcol)!=Br.get_p(Bcol+1)){ //not an empty column
253  //form the actual column of Ar
254  Sparseint Acol=Acdcol+AConditionColumns*contractcol;
255  if (Ar.get_p(Acol)!=Ar.get_p(Acol+1)){ //don't mess about with empty columns
256  for (Sparseint Bp=Br.get_p(Bcol);Bp<Br.get_p(Bcol+1);++Bp){ //rows of Br
257  for (Sparseint Ap=Ar.get_p(Acol);Ap<Ar.get_p(Acol+1);++Ap){ //rows of Ar
258  //no need to conjugate as the reshapes did that already
259  fullsparse.entry(Ar.get_i(Ap)+Ar.rows()*Br.get_i(Bp),Acdcol+AConditionColumns*Bcdcol,Ar.get_x(Ap)*Br.get_x(Bp));
260  }
261  }
262  }
263  }
264  }
265  }
266  }
267  }
268  fullsparse.finalise();
269  //NEEDS to figure out correct columns by charges!!!!!
270 
271  return TranslationBlock<T>(fullsparse,extractioncolumns,extractioncolumns);
272  }
